def is_prime(n: int) -> bool:
    if n < 2: return False
    if n == 2: return True
    if n % 2 == 0: return False
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0: return False
    return True

使用例

prime_factors(n)

素因数分解 O(√n) → {素因数: 指数}

def prime_factors(n: int) -> dict[int, int]:
    factors = defaultdict(int)
    d = 2
    while d * d <= n:
        while n % d == 0:
            factors[d] += 1
            n //= d
        d += 1
    if n > 1:
        factors[n] += 1
    return dict(factors)

使用例

factors = prime_factors(24)  # {2: 3, 3: 1} (24 = 2^3 * 3)

divisors(n)

約数列挙 O(√n) → ソート済みリスト

def divisors(n: int) -> list[int]:
    divs = []
    for i in range(1, int(n**0.5) + 1):
        if n % i == 0:
            divs.append(i)
            if i != n // i:
                divs.append(n // i)
    return sorted(divs)

使用例

sieve(n)

エラトステネスの篩 O(n log log n)

def sieve(n: int) -> tuple[list[bool], list[int]]:
    is_prime_arr = [True] * (n + 1)
    if n >= 0: is_prime_arr[0] = False
    if n >= 1: is_prime_arr[1] = False
    for p in range(2, int(n ** 0.5) + 1):
        if is_prime_arr[p]:
            for q in range(p * p, n + 1, p):
                is_prime_arr[q] = False
    primes = [i for i in range(2, n + 1) if is_prime_arr[i]]
    return is_prime_arr, primes

使用例

# primes = [2, 3, 5, 7, 11, 13, ...]

def ext_gcd(a: int, b: int) -> tuple[int, int, int]:
    if b == 0:
        return a, 1, 0
    g, x, y = ext_gcd(b, a % b)
    return g, y, x - (a // b) * y

pow_mod(x, n, mod)

高速べき乗 O(log n)

def pow_mod(x: int, n: int, mod: int = MOD) -> int:
    res = 1
    x %= mod
    while n > 0:
        if n & 1:
            res = res * x % mod
        x = x * x % mod
        n >>= 1
    return res

使用例

prefix_sum(arr)

1次元累積和 → ps[r] - ps[l] で [l, r) の和

def prefix_sum(arr: list[int]) -> list[int]:
    ps = [0]
    for x in arr:
        ps.append(ps[-1] + x)
    return ps

使用例

# [l, r) の和は ps[r] - ps[l]
sum_2_4 = ps[4] - ps[2]  # A[2] + A[3] = 3 + 4 = 7

prefix_sum_2d(grid)

2次元累積和

def prefix_sum_2d(grid: list[list[int]]) -> list[list[int]]:
    H, W = len(grid), len(grid[0])
    ps = [[0] * (W + 1) for _ in range(H + 1)]
    for i in range(H):
        for j in range(W):
            ps[i+1][j+1] = ps[i][j+1] + ps[i+1][j] - ps[i][j] + grid[i][j]
    return ps

使用例

grid = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]

# range_sum_2d で矩形和を取得

range_sum_2d(ps, y1, x1, y2, x2)

2次元累積和から [y1, y2) × [x1, x2) の和を取得

def range_sum_2d(ps: list[list[int]], y1: int, x1: int, y2: int, x2: int) -> int:
    return ps[y2][x2] - ps[y1][x2] - ps[y2][x1] + ps[y1][x1]

class Imos1D:
    def __init__(self, n: int):
        self.n = n
        self.diff = [0] * (n + 1)
    
    def add(self, l: int, r: int, x: int = 1) -> None:
        """[l, r) に x を加算"""

        self.diff[r] -= x

        """累積和を計算して結果を返す"""

bfs(g, s)

BFS最短距離（重みなし）→ 到達不能は-1

def bfs(g: list[list[int]], s: int) -> list[int]:
    dist = [-1] * len(g)
    dist[s] = 0
    q = deque([s])
    while q:
        v = q.popleft()
        for to in g[v]:
            if dist[to] == -1:
                dist[to] = dist[v] + 1
                q.append(to)
    return dist

使用例

def bfs_grid(grid: list[list[str]], sy: int, sx: int, wall: str = '#') -> list[list[int]]:
    H, W = len(grid), len(grid[0])
    dist = [[-1] * W for _ in range(H)]
    dist[sy][sx] = 0
    q = deque([(sy, sx)])
    while q:

dijkstra(g, s)

Dijkstra法 O((V+E)log V)

def dijkstra(g: list[list[tuple[int, int]]], s: int) -> list[int]:
    dist = [INF] * len(g)
    dist[s] = 0
    pq = [(0, s)]
    while pq:
        d, v = heapq.heappop(pq)
        if d > dist[v]: continue
        for to, w in g[v]:
            if dist[v] + w < dist[to]:
                dist[to] = dist[v] + w
                heapq.heappush(pq, (dist[to], to))
    return dist

bellman_ford(n, edges, s)

Bellman-Ford法 O(VE) → (距離, 負閉路あり?)

def bellman_ford(n: int, edges: list[tuple[int, int, int]], s: int) -> tuple[list[int], bool]:
    dist = [INF] * n
    dist[s] = 0
    for i in range(n):
        updated = False
        for u, v, w in edges:
            if dist[u] != INF and dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                updated = True
        if not updated: break
        if i == n - 1: return dist, True
    return dist, False

使用例

warshall_floyd(n, edges)

Warshall-Floyd法 O(V³) → 全点対最短距離

def warshall_floyd(n: int, edges: list[tuple[int, int, int]]) -> list[list[int]]:
    dist = [[INF] * n for _ in range(n)]
    for i in range(n): dist[i][i] = 0
    for u, v, w in edges:
        dist[u][v] = min(dist[u][v], w)
    for k in range(n):
        for i in range(n):
            for j in range(n):
                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])
    return dist

使用例

def topological_sort(g: list[list[int]]) -> list[int] | None:
    n = len(g)
    indeg = [0] * n
    for v in range(n):
        for to in g[v]:
            indeg[to] += 1
    q = deque([i for i in range(n) if indeg[i] == 0])
    result = []
    while q:
        v = q.popleft()
        result.append(v)
        for to in g[v]:
            indeg[to] -= 1
            if indeg[to] == 0:
                q.append(to)
    return result if len(result) == n else None

def tree_diameter(g: list[list[int]]) -> tuple[int, int, int]:
    def bfs_farthest(s: int) -> tuple[int, int]:
        dist = bfs(g, s)
        farthest = max(range(len(g)), key=lambda x: dist[x])
        return farthest, dist[farthest]
    
    u, _ = bfs_farthest(0)
    v, d = bfs_farthest(u)
    return d, u, v

tree_parent(g, root)

各頂点の親を計算（根は-1）

def tree_parent(g: list[list[int]], root: int = 0) -> list[int]:
    n = len(g)
    parent = [-1] * n
    visited = [False] * n
    visited[root] = True
    q = deque([root])
    while q:
        v = q.popleft()
        for to in g[v]:
            if not visited[to]:
                visited[to] = True
                parent[to] = v
                q.append(to)
    return parent

subtree_size(g, root)

部分木のサイズを計算

def subtree_size(g: list[list[int]], root: int = 0) -> list[int]:
    n = len(g)
    size = [1] * n
    parent = tree_parent(g, root)
    depth = tree_depth(g, root)
    order = sorted(range(n), key=lambda x: -depth[x])
    for v in order:
        if parent[v] != -1:
            size[parent[v]] += size[v]
    return size

def z_algorithm(s: str) -> list[int]:
    n = len(s)
    z = [0] * n
    z[0] = n
    i, j = 1, 0
    while i < n:
        while i + j < n and s[j] == s[i + j]:
            j += 1
        z[i] = j
        # ...
    return z

run_length_encode(s)

ランレングス圧縮 → [(文字, 連続数), ...]

def run_length_encode(s: str | list) -> list[tuple]:
    if not s: return []
    result = []
    current, count = s[0], 1
    for i in range(1, len(s)):
        if s[i] == current:
            count += 1
        else:
            result.append((current, count))
            current, count = s[i], 1
    result.append((current, count))
    return result

使用例

rle = run_length_encode("aaabbcccc")
# [('a', 3), ('b', 2), ('c', 4)]

binary_search_min(ng, ok, check)

条件を満たす最小値を探す

def binary_search_min(ng: int, ok: int, check: Callable[[int], bool]) -> int:
    """check(x) = True となる最小の x を返す
    ng: 条件を満たさない値, ok: 条件を満たす値"""
    while abs(ok - ng) > 1:
        mid = (ok + ng) // 2
        if check(mid):
            ok = mid
        else:
            ng = mid
    return ok

使用例

def binary_search_float(ok: float, ng: float, check: Callable[[float], bool], iterations: int = 100) -> float:
    for _ in range(iterations):

def compress(arr: list[int]) -> tuple[dict[int, int], list[int]]:
    xs = sorted(set(arr))
    mp = {x: i for i, x in enumerate(xs)}
    return mp, xs
